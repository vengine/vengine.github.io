258  void PushQuantity(MSSymbolValue quantity)
259  {
260  MSQuantityValue quantityValue;
261  if (quantity.valueID == MS_IMMEDIATE_NUMBER_VALUE) {
262  quantityValue.quantityType = IMMEDIATE_NUMBER;
263  MSImmediateNumberValue* immeNumberValue =
264  (MSImmediateNumberValue*)&m_valueStack[quantity.valueStackPointer];
265  quantityValue.number = *immeNumberValue;
266  }
267  else if (quantity.valueID == MS_STRING_VALUE) {
268  quantityValue.quantityType = STRING;
269  MSStringValue* strValue =
270  (MSStringValue*)&m_valueStack[quantity.valueStackPointer];
271  quantityValue.string = strValue->string;
272  }
273  else if (quantity.valueID == MS_VARIABLE_VALUE) {
274  quantityValue.quantityType = VARIABLE;
275  MSVariableValue* varValue =
276  (MSVariableValue*)&m_valueStack[quantity.valueStackPointer];
277  quantityValue.variable = *varValue;
278  }
279  else if (quantity.valueID == MS_QUANTITY_GROUP_VALUE) {
280  quantityValue.quantityType = QUANTITY_VALUE_GROUP;
281  quantityValue.quantityValueGroupIndex = quantity.valueStackPointer;
282  }
283  else {
284  VEngineExce(SymbolValueInvalidTypeException,
285  "push quantity target type must be immediate number or quantity group");
286  }
287  EAssert(m_tempQuantityGroupValueStack.size(), "temp quantity group value stack is empty!");
288  MSQuantityGroupValue* tempQuantityGroupValue = m_tempQuantityGroupValueStack.back();
289  tempQuantityGroupValue->quantities[tempQuantityGroupValue->numQuantities] =
290  quantityValue;
291  tempQuantityGroupValue->numQuantities++;
292  }

304  {
305  MSQuantityGroupValue* tempQuantityGroupValue =
306  m_tempQuantityGroupValueStack.back();
307  if (tempQuantityGroupValue->numQuantities > 2) {
308  MSQuantityValue secondLastQuantity = tempQuantityGroupValue->quantities[0];
309  MSQuantityValue lastQuantity = tempQuantityGroupValue->quantities[1];
310  int j = 0;
311  for (int i = tempQuantityGroupValue->numQuantities - 1; i >= 2; i--, j++) {
312  tempQuantityGroupValue->quantities[j] =
313  tempQuantityGroupValue->quantities[i];
314  }
315  tempQuantityGroupValue->quantities[tempQuantityGroupValue->numQuantities - 2] =
316  secondLastQuantity;
317  tempQuantityGroupValue->quantities[tempQuantityGroupValue->numQuantities - 1] =
318  lastQuantity;
319  }
320  int stackPointer = Push(tempQuantityGroupValue,
321  sizeof(*tempQuantityGroupValue) +
322  sizeof(MSQuantityValue) * tempQuantityGroupValue->numQuantities);
323  tempQuantityGroupValue->numQuantities = 0;
324  m_tempQuantityGroupValuePool.push_back(tempQuantityGroupValue);
325  m_tempQuantityGroupValueStack.pop_back();
326  MSSymbolValue ret = {MS_QUANTITY_GROUP_VALUE, stackPointer};
327  return ret;
